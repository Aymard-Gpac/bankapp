Examples
// Création réussie d'un client
const result = await ClientService.createClientWithDefaultAccounts({
  first_name: "Jean",
  last_name: "Dupont",
  email: "jean.dupont@email.com",
  address: "123 rue Principale",
  city: "Montréal",
  state: "QC",
  postal_code: "H1H 1H1",
  date_of_birth: "1990-01-01",
  password: "Password123!" // optionnel
});

if (result.ok) {
  console.log(result.data.message); // "Client created + 3 accounts created"
  console.log(`Client ID: ${result.data.userId}`);
} else {
  console.log(`Erreur ${result.status}: ${result.error}`);
}
// Création avec mot de passe généré automatiquement
const result = await ClientService.createClientWithDefaultAccounts({
  first_name: "Marie",
  last_name: "Lambert",
  email: "marie@email.com"
  // password non fourni → généré automatiquement
});
// Erreur: email déjà existant
const result = await ClientService.createClientWithDefaultAccounts({
  first_name: "Pierre",
  last_name: "Tremblay",
  email: "existant@email.com" // déjà en BD
});
// result = { ok: false, status: 409, error: "Email already exists" }
